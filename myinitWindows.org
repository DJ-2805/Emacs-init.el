    :ensure t)
  (ac-config-default)
  (global-auto-complete-mode t)
  (add-to-list 'ac-modes 'c++-mode)

  ;; turn on semantic
  (semantic-mode 1)

  ;; define a function which adds semantic as backup to auto-complete
  (defun my:add-semantic-to-autocomplete()
